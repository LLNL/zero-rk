
cmake_minimum_required(VERSION 3.10)
project(sundials5_build)

include(ExternalProject)
ExternalProject_Add(
  sundials5
  URL        https://github.com/LLNL/sundials/archive/refs/tags/v5.8.0.tar.gz
  URL_HASH   MD5=a087032a72127c12e162bfe4405b03cd
  CMAKE_ARGS -DCMAKE_INSTALL_PREFIX=${sundials_prefix}
             -DEXAMPLES_ENABLE_C=OFF
             -DEXAMPLES_INSTALL=OFF
             -DMPI_ENABLE=${ENABLE_MPI}
             -DCMAKE_BUILD_TYPE=${CMAKE_BUILD_TYPE}
             -DCMAKE_C_COMPILER=${MPI_C_COMPILER}
             -DCMAKE_POSITION_INDEPENDENT_CODE=ON
)
